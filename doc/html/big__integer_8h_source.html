350 std::strong_ordering compare(const biginteger& rhs) const {
351 // -a, +b
352 if (this->is_negative() && rhs.is_positive()) {
353 return std::strong_ordering::less;
354 }
355
356 // +a, -b
357 if (this->is_positive() && rhs.is_negative()) {
358 return std::strong_ordering::greater;
359 }
360
361 // -a, -b
362 if (this->is_negative() && rhs.is_negative()) {
363 if (this->value_.length() == rhs.value_.length()) {
364 if (this->value_ > rhs.value_) {
365 return std::strong_ordering::less;
366 }
367 return (this->value_ == rhs.value_)
368 ? std::strong_ordering::equivalent
369 : std::strong_ordering::greater;
370 }
371 return (this->value_.length() < rhs.value_.length())
372 ? std::strong_ordering::greater
373 : std::strong_ordering::less;
374 }
375

383 : std::strong_ordering::less;

427 [[nodiscard]]
428 biginteger add(const biginteger& rhs) const {
429 // (-a)+(+b)
430 if (this->is_negative() && rhs.is_positive()) {
431 return rhs.subtract(this->negate());
432 }
433
434 // (+a)+(-b)
435 if (this->is_positive() && rhs.is_negative()) {
436 return this->subtract(rhs.negate());
437 }
438
439 // (+a)+(+b) or (-a)+(-b)
440 auto lhs_value = this->value_;
441 auto rhs_value = rhs.value_;
442
443 auto diff_length = std::abs(
444 static_cast<int>(lhs_value.length() - rhs_value.length()));
445
446 const auto min_value{'0'};
447 const auto max_value{'9'};
448
449 if (lhs_value.length() > rhs_value.length()) {
450 rhs_value.insert(0, diff_length, min_value);
451 } else {
452 lhs_value.insert(0, diff_length, min_value);
453 }
454 std::ranges::reverse(lhs_value);
455 std::ranges::reverse(rhs_value);
456
457 usize index{};
458 char carry{min_value};
459
460 for (char& lhs_char : lhs_value) {
461 lhs_char = static_cast<char>(
462 (carry - min_value) +
463 (lhs_char - min_value) +
464 (rhs_value.at(index) - min_value) +
465 min_value);
466
467 if (lhs_char > max_value) {
468 lhs_char -= 10;
469 carry = (min_value + 1);
470 } else {
471 carry = min_value;
472 }
473 ++index;
474 }
475
476 if (carry > min_value) {
477 lhs_value.append(1, carry);
478 }
479 std::ranges::reverse(lhs_value);
480
481 return this->is_negative() ? value_from(lhs_value).negate() : std::move(lhs_value);
482 }

489 [[nodiscard]]
490 biginteger subtract(const biginteger& rhs) const {
491 // (-a)-(+b) or (+a)-(-b)
492 if ((this->is_negative() && rhs.is_positive()) ||
493 (this->is_positive() && rhs.is_negative()))
494 {
495 return this->add(rhs.negate());
496 }
497
498 // (+a)-(+b) or (-a)-(-b)
499 if (this->is_negative()) {
500 return this->add(rhs.negate());
501 }
502 bool inverted_sign = (this->compare(rhs) == std::strong_ordering::less);
503
504 std::string subtracted = inverted_sign ? rhs.value_ : this->value_;
505 std::string removed = inverted_sign ? this->value_ : rhs.value_;
506
507 auto diff_length = std::abs(
508 static_cast<int>(subtracted.length() - removed.length()));
509
510 const auto min_value{'0'};
511
512 if (subtracted.size() > removed.size()) {
513 removed.insert(0, diff_length, min_value);
514 } else {
515 subtracted.insert(0, diff_length, min_value);
516 }
517 std::ranges::reverse(subtracted);
518 std::ranges::reverse(removed);
519
520 usize index{};
521
522 for (char& symbol : subtracted) {
523 if (symbol < removed.at(index)) {
524 symbol += 10;
525 --subtracted.at(index + 1);
526 }
527 symbol = static_cast<char>(
528 (symbol - min_value) - (removed.at(index) - min_value) + min_value);
529
530 ++index;
531 }
532 std::ranges::reverse(subtracted);
533
534 while (subtracted.front() == min_value && subtracted.length() != 1) {
535 subtracted.erase(0, 1);
536 }
537 return inverted_sign ? value_from(subtracted).negate() : std::move(subtracted);
538 }

545 [[nodiscard]]
546 biginteger multiply(const biginteger& rhs) const {
547 auto lhs_value = this->value_;
548 auto rhs_value = rhs.value_;
549
550 std::ranges::reverse(lhs_value);
551 std::ranges::reverse(rhs_value);
552
553 const auto min_value{'0'};
554 const auto max_value{'9'};
555
556 int index{};
557 char carry{min_value};
558
559 auto temp = value_from(0);
560 uint8 result;
561
562 for (const auto& lhs_char : lhs_value) {
563 auto operation = std::string{}.insert(0, index, min_value);
564
565 for (const auto& rhs_char : rhs_value) {
566 result =
567 ((lhs_char - min_value) * (rhs_char - min_value)) +
568 (carry - min_value) +
569 min_value;
570
571 carry = min_value;
572
573 if (result > max_value) {
574 while (result > max_value) {
575 result -= 10;
576 ++carry;
577 }
578 }
579 operation.insert(0, 1, static_cast<char>(result));
580 }
581 if (carry > min_value) {
582 operation.insert(0, 1, carry);
583 carry = min_value;
584 }
585 temp += operation;
586 ++index;
587 }
588 auto positive =
589 (this->is_negative() && rhs.is_negative()) ||
590 (this->is_positive() && rhs.is_positive());
591
592 return positive ? std::move(temp) : temp.negate();
593 }

600 [[nodiscard]]
601 biginteger divide(const biginteger& rhs) const {
602 if (rhs.equal(0)) {
603 throw std::invalid_argument("divide by zero");
604 }
605 if (rhs.equal(1)) {
606 return *this;
607 }
608 if (this->equal(rhs)) {
609 return 1;
610 }
611 auto lhs_value = this->value_;
612 std::ranges::reverse(lhs_value);
613
614 auto rhs_abs = rhs.abs();
615
616 std::string lhs_quotient;
617 std::string rhs_quotient;
618
619 do {
620 lhs_quotient.push_back(lhs_value.back());
621 lhs_value.pop_back();
622
623 biginteger dividend_value{lhs_quotient};
624
625 if (dividend_value < rhs_abs) {
626 rhs_quotient.push_back('0');
627 continue;
628 }
629 auto number = value_from(2);
630
631 while (rhs_abs.multiply(number) <= dividend_value) {
632 ++number;
633 }
634 --number;
635 rhs_quotient.append(number.to_string());
636 lhs_quotient = dividend_value.subtract(rhs_abs.multiply(number)).to_string();
637 } while (!lhs_value.empty());
638
639 auto positive =
640 (this->is_negative() && rhs.is_negative()) ||
641 (this->is_positive() && rhs.is_positive());
642
643 return positive ? std::move(rhs_quotient) : value_from(rhs_quotient).negate();
644 }

739 biginteger pow(const biginteger& rhs) const {
740 if (rhs.equal(0)) {
741 return 1;
742 }
743 if (this->equal(0)) {
744 return 0;
745 }
746 if (rhs.compare(0) == std::strong_ordering::less) {
747 return this->abs().equal(1) ? *this : 0;
748 }
749 auto init_exp{rhs};
750 auto result{*this};
751 auto result_odd = value_from(1);
752
753 while (init_exp.compare(1) == std::strong_ordering::greater) {
754 if (init_exp.mod(2)) {
755 result_odd *= result;
756 }
757 result *= result;
758 init_exp /= 2;
759 }
760 return result.multiply(result_odd);

775 biginteger sqrt() const {
776 if (this->compare(0) == std::strong_ordering::less) {
777 throw std::invalid_argument("sqrt of a negative value");
778 }
779
780 // Small numbers optimisation.
781 if (this->equal(0)) {
782 return 0;
783 }
784 if (this->compare(4) == std::strong_ordering::less) {
785 return 1;
786 }
787 if (this->compare(9) == std::strong_ordering::less) {
788 return 2;
789 }
790 if (this->compare(16) == std::strong_ordering::less) {
791 return 3;
792 }
793 auto previous = value_from(-1);
794 auto current = value_from(this->to_string().size() / 2 - 1).pow10();
795
796 while (current.subtract(previous).abs().compare(1) == std::strong_ordering::greater) {
797 previous = current;
798 current = this->divide(previous).add(previous).divide(2);
799 }

1050
