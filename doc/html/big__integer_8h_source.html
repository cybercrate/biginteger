132 // Remove leading zero.

134 rhs.erase(0, 1);

136 if (this->radix_ == 10) {
137 this->value_ = rhs;
138 return *this;

140 // Convert to base ten to use trivial algorithm.
141 char last_char = rhs.back();
142 int last = ('0' <= last_char) && (last_char <= '9')
143 ? (last_char - '0')

146 auto value_length = static_cast<std::int64_t>(rhs.length());
147
148 big_integer power{value_length - 1};
149 big_integer converted_value{last};

152 for (int i = 0; i < value_length - 1; i++) {

155 ? (c - '0')
156 : (std::tolower(c) - 'a') + 10;

158 converted_value += big_integer{current_value}.multiply(radix_value.pow(power--));

357 [[nodiscard]]
358 big_integer add(const big_integer& rhs) const {
359 // (-a)+(+b)
360 if (this->is_negative() && rhs.is_positive())
361 return rhs.subtract(this->negate());
362
363 // (+a)+(-b)
364 if (this->is_positive() && rhs.is_negative())
365 return this->subtract(rhs.negate());
366
367 // (+a)+(+b) or (-a)+(-b)
368 auto lhs_value = this->value_;
369 auto rhs_value = rhs.value_;
370
371 auto diff_length = std::abs(static_cast<int>(lhs_value.length() - rhs_value.length()));
372
373 if (lhs_value.length() > rhs_value.length())
374 rhs_value.insert(0, diff_length, '0');
375 else
376 lhs_value.insert(0, diff_length, '0');
377
378 std::ranges::reverse(lhs_value);
379 std::ranges::reverse(rhs_value);
380
381 int index{};
382 char carry{'0'};
383
384 for (char& lhs_char : lhs_value) {
385 lhs_char = static_cast<char>(
386 (carry - '0') + (lhs_char - '0') + (rhs_value.at(index) - '0') + '0');
387
388 if (lhs_char > '9') {
389 lhs_char -= 10;
390 carry = '1';
391 } else {
392 carry = '0';
393 }
394 index++;
395 }
396 if (carry > '0')
397 lhs_value.append(1, carry);
398
399 std::ranges::reverse(lhs_value);
400
401 return this->is_negative() ? big_integer{lhs_value}.negate() : lhs_value;
402 }

409 [[nodiscard]]
410 big_integer subtract(const big_integer& rhs) const {
411 // (-a)-(+b) or (+a)-(-b)
412 if ((this->is_negative() && rhs.is_positive()) ||
413 (this->is_positive() && rhs.is_negative()))
414 {
415 return this->add(rhs.negate());
416 }
417
418 // (+a)-(+b) or (-a)-(-b)
419 if (this->is_negative())
420 return this->add(rhs.negate());
421
422 bool inverted_sign = (this->compare(rhs) == std::strong_ordering::less);
423
424 std::string subtracted = inverted_sign ? rhs.value_ : this->value_;
425 std::string removed = inverted_sign ? this->value_ : rhs.value_;
426
427 auto diff_length = std::abs(static_cast<int>(subtracted.length() - removed.length()));
428
429 if (subtracted.size() > removed.size())
430 removed.insert(0, diff_length, '0');
431 else
432 subtracted.insert(0, diff_length, '0');
433
434 std::ranges::reverse(subtracted);
435 std::ranges::reverse(removed);
436
437 int index{};
438
439 for (char& symbol : subtracted) {
440 if (symbol < removed.at(index)) {
441 symbol += 10;
442 subtracted[index + 1]--;
443 }
444 symbol = static_cast<char>((symbol - '0') - (removed.at(index) - '0') + '0');
445 index++;
446 }
447 std::ranges::reverse(subtracted);
448
449 while (subtracted.front() == '0' && subtracted.length() != 1)
450 subtracted.erase(0, 1);
451
452 return inverted_sign ? big_integer{subtracted}.negate() : subtracted;

460 [[nodiscard]]
461 big_integer multiply(const big_integer& rhs) const {
462 auto lhs_value = this->value_;
463 auto rhs_value = rhs.value_;
464
465 std::ranges::reverse(lhs_value);
466 std::ranges::reverse(rhs_value);
467
468 int step{};
469 char carry{'0'};
470
471 big_integer temp;
472
473 for (const auto& lhs_char : lhs_value) {
474 auto operation = std::string{}.insert(0, step, '0');
475
476 for (const auto& rhs_char : rhs_value) {
477 std::uint8_t result = ((lhs_char - '0') * (rhs_char - '0')) + (carry - '0') + '0';
478 carry = '0';
479
480 if (result > '9') {
481 while (result > '9') {
482 result -= 10;
483 carry++;
484 }
485 }
486 operation.insert(0, 1, static_cast<char>(result));
487 }
488 if (carry > '0') {
489 operation.insert(0, 1, carry);
490 carry = '0';
491 }
492 temp += operation;
493 step++;
494 }
495 auto positive = (this->is_negative() && rhs.is_negative()) ||
496 (this->is_positive() && rhs.is_positive());
497 return positive ? temp :temp.negate();

505 [[nodiscard]]
506 big_integer divide(const big_integer& rhs) const {
507 if (rhs.equal(0))
508 throw std::invalid_argument("divide by zero");
509
510 if (rhs.equal(1))
511 return *this;
512
513 if (this->equal(rhs))
514 return 1;
515
516 auto lhs_value = this->value_;
517 std::ranges::reverse(lhs_value);
518
519 auto rhs_abs = rhs.abs();
520
521 std::string lhs_quotient;
522 std::string rhs_quotient;
523
524 do {
525 lhs_quotient.push_back(lhs_value.back());
526 lhs_value.pop_back();
527
528 big_integer dividend_value{lhs_quotient};
529
530 if (dividend_value < rhs_abs) {
531 rhs_quotient.push_back('0');
532 continue;
533 }
534 auto number{value_from<2>()};
535
536 while (rhs_abs.multiply(number) <= dividend_value)
537 number++;
538
539 number--;
540 rhs_quotient.append(number.to_string());
541 lhs_quotient = dividend_value.subtract(rhs_abs.multiply(number)).to_string();
542 } while (!lhs_value.empty());
543
544 bool positive = (this->is_negative() && rhs.is_negative()) ||
545 (this->is_positive() && rhs.is_positive());
546 return positive ? rhs_quotient : big_integer{rhs_quotient}.negate();
547 }

626 [[nodiscard]]
627 big_integer pow(const big_integer& rhs) const {
628 if (rhs.equal(0))
629 return 1;
630
631 if (rhs.compare(0) == std::strong_ordering::less) {
632 if (this->equal(0))
633 throw std::logic_error("");
634
635 return this->abs().equal(1) ? *this : value_from<0>();
636 }
637 auto init_exp{rhs};
638 auto result{*this};
639 auto result_odd{value_from<1>()};
640
641 while (init_exp.compare(1) == std::strong_ordering::greater) {
642 if (init_exp.mod(2)) result_odd *= result;
643 result *= result;
644 init_exp /= 2;
645 }
646 return result.multiply(result_odd);
647 }

665 big_integer abs_lhs = this->abs();
666 big_integer abs_rhs = rhs.abs();
667
668 if (abs_rhs.equal(0) || abs_lhs.equal(0))
669 return 0;
670
671 auto remainder{abs_rhs};
672
673 while (!remainder.equal(0)) {
674 remainder = abs_lhs.mod(abs_rhs);
675 abs_lhs = abs_rhs; // previous remainder
676 abs_rhs = remainder; // current remainder
677 }
678 return std::move(abs_lhs);

701 [[nodiscard]]
702 std::strong_ordering compare(const big_integer& rhs) const {
703 // -a, +b
704 if (this->is_negative() && rhs.is_positive())
705 return std::strong_ordering::less;
706
707 // +a, -b
708 if (this->is_positive() && rhs.is_negative())
709 return std::strong_ordering::greater;
710
711 // -a, -b
712 if (this->is_negative() && rhs.is_negative()) {
713 if (this->value_.length() == rhs.value_.length()) {
714 if (this->value_ > rhs.value_)
715 return std::strong_ordering::less;
716
717 return (this->value_ == rhs.value_)
718 ? std::strong_ordering::equivalent
719 : std::strong_ordering::greater;
720 }
721 return (this->value_.length() < rhs.value_.length())
722 ? std::strong_ordering::greater
723 : std::strong_ordering::less;
724 }
725
726 // +a, +b
727 if (this->is_positive() && rhs.is_positive()) {
728 if (this->value_.length() == rhs.value_.length()) {
729 if (this->value_ > rhs.value_)
730 return std::strong_ordering::greater;
731
732 return (this->value_ == rhs.value_)
733 ? std::strong_ordering::equivalent
734 : std::strong_ordering::less;
735 }
736 return (this->value_.length() < rhs.value_.length())
737 ? std::strong_ordering::less
738 : std::strong_ordering::greater;
739 }
740
741 return std::strong_ordering::equivalent;
