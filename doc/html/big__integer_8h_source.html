362 [[nodiscard]]
363 std::strong_ordering compare(const big_integer& rhs) const
364 {
365 // -a, +b
366 if (this->is_negative() && rhs.is_positive())
367 return std::strong_ordering::less;
368
369 // +a, -b
370 if (this->is_positive() && rhs.is_negative())
371 return std::strong_ordering::greater;
372
373 // -a, -b
374 if (this->is_negative() && rhs.is_negative()) {
375 if (this->value_.length() == rhs.value_.length()) {
376 if (this->value_ > rhs.value_)
377 return std::strong_ordering::less;
378
379 return (this->value_ == rhs.value_)
380 ? std::strong_ordering::equivalent
381 : std::strong_ordering::greater;
382 }
383 return (this->value_.length() < rhs.value_.length())
384 ? std::strong_ordering::greater
385 : std::strong_ordering::less;
386 }
387
388 // +a, +b
389 if (this->value_.length() == rhs.value_.length()) {
390 if (this->value_ > rhs.value_)
391 return std::strong_ordering::greater;
392
393 return (this->value_ == rhs.value_)
394 ? std::strong_ordering::equivalent
395 : std::strong_ordering::less;
396 }
397 return (this->value_.length() < rhs.value_.length())
398 ? std::strong_ordering::less
399 : std::strong_ordering::greater;

442 [[nodiscard]]
443 big_integer add(const big_integer& rhs) const
444 {
445 // (-a)+(+b)
446 if (this->is_negative() && rhs.is_positive())
447 return rhs.subtract(this->negate());
448
449 // (+a)+(-b)
450 if (this->is_positive() && rhs.is_negative())
451 return this->subtract(rhs.negate());
452
453 // (+a)+(+b) or (-a)+(-b)
454 auto lhs_value = this->value_;
455 auto rhs_value = rhs.value_;
456
457 auto diff_length = std::abs(static_cast<int>(lhs_value.length() - rhs_value.length()));
458
459 if (lhs_value.length() > rhs_value.length())
460 rhs_value.insert(0, diff_length, '0');
461 else
462 lhs_value.insert(0, diff_length, '0');
463
464 std::ranges::reverse(lhs_value);
465 std::ranges::reverse(rhs_value);
466
467 std::size_t index{};
468 char carry{'0'};
469
470 for (char& lhs_char : lhs_value) {
471 lhs_char = static_cast<char>(
472 (carry - '0') + (lhs_char - '0') + (rhs_value.at(index) - '0') + '0');
473
474 if (lhs_char > '9') {
475 lhs_char -= 10;
476 carry = '1';
477 }
478 else {
479 carry = '0';
480 }
481 ++index;
482 }
483
484 if (carry > '0')
485 lhs_value.append(1, carry);
486
487 std::ranges::reverse(lhs_value);
488
489 return this->is_negative() ? value_from(lhs_value).negate() : std::move(lhs_value);
490 }

497 [[nodiscard]]
498 big_integer subtract(const big_integer& rhs) const
499 {
500 // (-a)-(+b) or (+a)-(-b)
501 if ((this->is_negative() && rhs.is_positive()) ||
502 (this->is_positive() && rhs.is_negative()))
503 {
504 return this->add(rhs.negate());
505 }
506
507 // (+a)-(+b) or (-a)-(-b)
508 if (this->is_negative())
509 return this->add(rhs.negate());
510
511 bool inverted_sign = (this->compare(rhs) == std::strong_ordering::less);
512
513 std::string subtracted = inverted_sign ? rhs.value_ : this->value_;
514 std::string removed = inverted_sign ? this->value_ : rhs.value_;
515
516 auto diff_length = std::abs(static_cast<int>(subtracted.length() - removed.length()));
517
518 if (subtracted.size() > removed.size())
519 removed.insert(0, diff_length, '0');
520 else
521 subtracted.insert(0, diff_length, '0');
522
523 std::ranges::reverse(subtracted);
524 std::ranges::reverse(removed);
525
526 std::size_t index{};
527
528 for (char& symbol : subtracted) {
529 if (symbol < removed.at(index)) {
530 symbol += 10;
531 subtracted[index + 1]--;
532 }
533 symbol = static_cast<char>((symbol - '0') - (removed.at(index) - '0') + '0');
534 ++index;
535 }
536 std::ranges::reverse(subtracted);
537
538 while (subtracted.front() == '0' && subtracted.length() != 1)
539 subtracted.erase(0, 1);
540
541 return inverted_sign ? value_from(subtracted).negate() : std::move(subtracted);
542 }

549 [[nodiscard]]
550 big_integer multiply(const big_integer& rhs) const
551 {
552 auto lhs_value = this->value_;
553 auto rhs_value = rhs.value_;
554
555 std::ranges::reverse(lhs_value);
556 std::ranges::reverse(rhs_value);
557
558 int index{};
559 char carry{'0'};
560
561 auto temp = value_from(0);
562
563 for (const auto& lhs_char : lhs_value) {
564 auto operation = std::string{}.insert(0, index, '0');
565
566 for (const auto& rhs_char : rhs_value) {
567 std::uint8_t result = ((lhs_char - '0') * (rhs_char - '0')) + (carry - '0') + '0';
568 carry = '0';
569
570 if (result > '9') {
571 while (result > '9') {
572 result -= 10;
573 ++carry;
574 }
575 }
576 operation.insert(0, 1, static_cast<char>(result));
577 }
578
579 if (carry > '0') {
580 operation.insert(0, 1, carry);
581 carry = '0';
582 }
583 temp += operation;
584 ++index;
585 }
586 auto positive = (this->is_negative() && rhs.is_negative()) ||
587 (this->is_positive() && rhs.is_positive());
588
589 return positive ? std::move(temp) : temp.negate();
590 }

597 [[nodiscard]]
598 big_integer divide(const big_integer& rhs) const
599 {
600 if (rhs.equal(0))
601 throw std::invalid_argument("divide by zero");
602
603 if (rhs.equal(1))
604 return *this;
605
606 if (this->equal(rhs))
607 return 1;
608
609 auto lhs_value = this->value_;
610 std::ranges::reverse(lhs_value);
611
612 auto rhs_abs = rhs.abs();
613
614 std::string lhs_quotient;
615 std::string rhs_quotient;
616
617 do {
618 lhs_quotient.push_back(lhs_value.back());
619 lhs_value.pop_back();
620
621 big_integer dividend_value{lhs_quotient};
622
623 if (dividend_value < rhs_abs) {
624 rhs_quotient.push_back('0');
625 continue;
626 }
627 auto number= value_from(2);
628
629 while (rhs_abs.multiply(number) <= dividend_value)
630 ++number;
631
632 --number;
633 rhs_quotient.append(number.to_string());
634 lhs_quotient = dividend_value.subtract(rhs_abs.multiply(number)).to_string();
635 } while (!lhs_value.empty());
636
637 bool positive = (this->is_negative() && rhs.is_negative()) ||
638 (this->is_positive() && rhs.is_positive());
639
640 return positive ? std::move(rhs_quotient) : value_from(rhs_quotient).negate();
641 }

743 [[nodiscard]]
744 big_integer pow(const big_integer& rhs) const
745 {
746 if (rhs.equal(0))
747 return 1;
748
749 if (this->equal(0))
750 return 0;
751
752 if (rhs.compare(0) == std::strong_ordering::less)
753 return this->abs().equal(1) ? *this : 0;
754
755 auto init_exp{rhs};
756 auto result{*this};
757 auto result_odd = value_from(1);
758
759 while (init_exp.compare(1) == std::strong_ordering::greater) {
760 if (init_exp.mod(2))
761 result_odd *= result;
762
763 result *= result;
764 init_exp /= 2;
765 }
766 return result.multiply(result_odd);
767 }

781 [[nodiscard]]
782 big_integer sqrt() const
783 {
784 if (this->compare(0) == std::strong_ordering::less)
785 throw std::invalid_argument("sqrt of a negative value");
786
787 // Small numbers optimisation.
788 if (this->equal(0))
789 return 0;
790
791 if (this->compare(4) == std::strong_ordering::less)
792 return 1;
793
794 if (this->compare(9) == std::strong_ordering::less)
795 return 2;
796
797 if (this->compare(16) == std::strong_ordering::less)
798 return 3;
799
800 auto previous = value_from(-1);
801 auto current = value_from(this->to_string().size() / 2 - 1).pow10();
802
803 while (current.subtract(previous).abs().compare(1) == std::strong_ordering::greater) {
804 previous = current;
805 current = this->divide(previous).add(previous).divide(2);
806 }
807 return std::move(current);
808 }

816 big_integer gcd(const big_integer& rhs) const
817 {
818 auto lhs_abs = this->abs();

820
821 if (rhs_abs.equal(0) || lhs_abs.equal(0))
822 return 0;
823

825
826 while (!remainder.equal(0)) {
827 remainder = lhs_abs.mod(rhs_abs);
828 lhs_abs = rhs_abs;
829 rhs_abs = remainder;
830 }

1083 // Converts string value to base ten.
1084 static std::string convert_to_base_ten(const std::string& value, radix_type radix)
1085 {
1086 char last_char = value.back();
1087 int last = ('0' <= last_char) && (last_char <= '9')
1088 ? (last_char - '0')
1089 : (tolower(last_char) - 'a') + 10;
1090
1091 auto value_length = static_cast<std::size_t>(value.length());
1092
1093 auto power = value_from(value_length - 1);
1094 auto converted_value = value_from(last);
1095 auto radix_value = value_from(static_cast<int>(radix));
1096
1097 for (int i = 0; i < value_length - 1; ++i) {
1098 auto c = value.at(i);
1099 auto current_value = ('0' <= c) && (c <= '9')
1100 ? (c - '0')
1101 : (std::tolower(c) - 'a') + 10;
1102
1103 converted_value += value_from(current_value).multiply(radix_value.pow(power--));
1104 }
1105 return std::move(converted_value.value_);
1106 }
