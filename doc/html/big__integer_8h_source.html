134 big_integer& operator=(std::string value) & {
135 if (this->is_negative())
136 value.erase(0, 1);
137
138 // Remove leading zero.
139 while (value.starts_with('0') && (value.length() != 1))
140 value.erase(0, 1);
141
142 if (this->radix_ == radix_type::decimal) {
143 this->value_ = value;
144 return *this;
145 }
146
147 // Convert to base ten to use trivial algorithm.
148 char last_char = value.back();
149 int last = ('0' <= last_char) && (last_char <= '9')
150 ? (last_char - '0')
151 : (tolower(last_char) - 'a') + 10;
152
153 auto value_length = static_cast<std::size_t>(value.length());
154
155 auto power = value_from(value_length - 1);
156 auto converted_value = value_from(last);
157 auto radix_value = value_from(static_cast<int>(this->radix_));
158
159 for (int i = 0; i < value_length - 1; ++i) {
160 auto c = value.at(i);
161 auto current_value = ('0' <= c) && (c <= '9')
162 ? (c - '0')
163 : (std::tolower(c) - 'a') + 10;
164
165 converted_value += value_from(current_value).multiply(radix_value.pow(power--));
166 }
167 *this = this->is_negative() ? converted_value.negate() : std::move(converted_value);
168 return *this;

344 [[nodiscard]]
345 std::strong_ordering compare(const big_integer& rhs) const {
346 // -a, +b
347 if (this->is_negative() && rhs.is_positive())
348 return std::strong_ordering::less;
349
350 // +a, -b
351 if (this->is_positive() && rhs.is_negative())
352 return std::strong_ordering::greater;
353
354 // -a, -b
355 if (this->is_negative() && rhs.is_negative()) {
356 if (this->value_.length() == rhs.value_.length()) {
357 if (this->value_ > rhs.value_)
358 return std::strong_ordering::less;
359
360 return (this->value_ == rhs.value_)
361 ? std::strong_ordering::equivalent
362 : std::strong_ordering::greater;
363 }
364 return (this->value_.length() < rhs.value_.length())
365 ? std::strong_ordering::greater
366 : std::strong_ordering::less;
367 }
368
369 // +a, +b
370 if (this->is_positive() && rhs.is_positive()) {
371 if (this->value_.length() == rhs.value_.length()) {
372 if (this->value_ > rhs.value_)
373 return std::strong_ordering::greater;
374
375 return (this->value_ == rhs.value_)
376 ? std::strong_ordering::equivalent
377 : std::strong_ordering::less;
378 }
379 return (this->value_.length() < rhs.value_.length())
380 ? std::strong_ordering::less
381 : std::strong_ordering::greater;
382 }
383
384 return std::strong_ordering::equivalent;
385 }

420 [[nodiscard]]
421 big_integer add(const big_integer& rhs) const {
422 // (-a)+(+b)
423 if (this->is_negative() && rhs.is_positive())
424 return rhs.subtract(this->negate());
425
426 // (+a)+(-b)
427 if (this->is_positive() && rhs.is_negative())
428 return this->subtract(rhs.negate());
429
430 // (+a)+(+b) or (-a)+(-b)
431 auto lhs_value = this->value_;
432 auto rhs_value = rhs.value_;
433
434 auto diff_length = std::abs(static_cast<int>(lhs_value.length() - rhs_value.length()));
435
436 if (lhs_value.length() > rhs_value.length())
437 rhs_value.insert(0, diff_length, '0');
438 else
439 lhs_value.insert(0, diff_length, '0');
440
441 std::ranges::reverse(lhs_value);
442 std::ranges::reverse(rhs_value);
443
444 std::size_t index{};
445 char carry{'0'};
446
447 for (char& lhs_char : lhs_value) {
448 lhs_char = static_cast<char>(
449 (carry - '0') + (lhs_char - '0') + (rhs_value.at(index) - '0') + '0');
450
451 if (lhs_char > '9') {
452 lhs_char -= 10;
453 carry = '1';
454 } else {
455 carry = '0';
456 }
457 ++index;
458 }
459 if (carry > '0')
460 lhs_value.append(1, carry);
461 std::ranges::reverse(lhs_value);
462
463 return this->is_negative() ? value_from(lhs_value).negate() : std::move(lhs_value);
464 }

471 [[nodiscard]]
472 big_integer subtract(const big_integer& rhs) const {
473 // (-a)-(+b) or (+a)-(-b)
474 if ((this->is_negative() && rhs.is_positive()) ||
475 (this->is_positive() && rhs.is_negative()))
476 {
477 return this->add(rhs.negate());
478 }
479
480 // (+a)-(+b) or (-a)-(-b)
481 if (this->is_negative())
482 return this->add(rhs.negate());
483
484 bool inverted_sign = (this->compare(rhs) == std::strong_ordering::less);
485
486 std::string subtracted = inverted_sign ? rhs.value_ : this->value_;
487 std::string removed = inverted_sign ? this->value_ : rhs.value_;
488
489 auto diff_length = std::abs(static_cast<int>(subtracted.length() - removed.length()));
490
491 if (subtracted.size() > removed.size())
492 removed.insert(0, diff_length, '0');
493 else
494 subtracted.insert(0, diff_length, '0');
495
496 std::ranges::reverse(subtracted);
497 std::ranges::reverse(removed);
498
499 std::size_t index{};
500
501 for (char& symbol : subtracted) {
502 if (symbol < removed.at(index)) {
503 symbol += 10;
504 subtracted[index + 1]--;
505 }
506 symbol = static_cast<char>((symbol - '0') - (removed.at(index) - '0') + '0');
507 ++index;
508 }
509 std::ranges::reverse(subtracted);
510
511 while (subtracted.front() == '0' && subtracted.length() != 1)
512 subtracted.erase(0, 1);
513
514 return inverted_sign ? value_from(subtracted).negate() : std::move(subtracted);
515 }

522 [[nodiscard]]
523 big_integer multiply(const big_integer& rhs) const {
524 auto lhs_value = this->value_;
525 auto rhs_value = rhs.value_;
526
527 std::ranges::reverse(lhs_value);
528 std::ranges::reverse(rhs_value);
529
530 int index{};
531 char carry{'0'};
532
533 auto temp = value_from(0);
534
535 for (const auto& lhs_char : lhs_value) {
536 auto operation = std::string{}.insert(0, index, '0');
537
538 for (const auto& rhs_char : rhs_value) {
539 std::uint8_t result = ((lhs_char - '0') * (rhs_char - '0')) + (carry - '0') + '0';
540 carry = '0';
541
542 if (result > '9') {
543 while (result > '9') {
544 result -= 10;
545 ++carry;
546 }
547 }
548 operation.insert(0, 1, static_cast<char>(result));
549 }
550 if (carry > '0') {
551 operation.insert(0, 1, carry);
552 carry = '0';
553 }
554 temp += operation;
555 ++index;
556 }
557 auto positive = (this->is_negative() && rhs.is_negative()) ||
558 (this->is_positive() && rhs.is_positive());
559 return positive ? std::move(temp) : temp.negate();
560 }

567 [[nodiscard]]
568 big_integer divide(const big_integer& rhs) const {
569 if (rhs.equal(0))
570 throw std::invalid_argument("divide by zero");
571
572 if (rhs.equal(1))
573 return *this;
574
575 if (this->equal(rhs))
576 return 1;
577
578 auto lhs_value = this->value_;
579 std::ranges::reverse(lhs_value);
580
581 auto rhs_abs = rhs.abs();
582
583 std::string lhs_quotient;
584 std::string rhs_quotient;
585
586 do {
587 lhs_quotient.push_back(lhs_value.back());
588 lhs_value.pop_back();
589
590 big_integer dividend_value{lhs_quotient};
591
592 if (dividend_value < rhs_abs) {
593 rhs_quotient.push_back('0');
594 continue;
595 }
596 auto number= value_from(2);
597
598 while (rhs_abs.multiply(number) <= dividend_value)
599 ++number;
600
601 --number;
602 rhs_quotient.append(number.to_string());
603 lhs_quotient = dividend_value.subtract(rhs_abs.multiply(number)).to_string();
604 } while (!lhs_value.empty());
605
606 bool positive = (this->is_negative() && rhs.is_negative()) ||
607 (this->is_positive() && rhs.is_positive());
608 return positive ? std::move(rhs_quotient) : value_from(rhs_quotient).negate();
609 }

687 [[nodiscard]]
688 big_integer pow(const big_integer& rhs) const {
689 if (rhs.equal(0))
690 return 1;
691
692 if (this->equal(0))
693 return 0;
694
695 if (rhs.compare(0) == std::strong_ordering::less)
696 return this->abs().equal(1) ? *this : 0;
697
698 auto init_exp{rhs};
699 auto result{*this};
700 auto result_odd = value_from(1);
701
702 while (init_exp.compare(1) == std::strong_ordering::greater) {
703 if (init_exp.mod(2)) result_odd *= result;
704 result *= result;
705 init_exp /= 2;
706 }
707 return result.multiply(result_odd);
708 }

719 [[nodiscard]]
720 big_integer sqrt() const {
721 if (this->compare(0) == std::strong_ordering::less)
722 throw std::invalid_argument("sqrt of a negative value");
723
724 // Small numbers optimisation.
725 if (this->equal(0)) return 0;
726 if (this->compare(4) == std::strong_ordering::less) return 1;
727 if (this->compare(9) == std::strong_ordering::less) return 2;
728 if (this->compare(16) == std::strong_ordering::less) return 3;
729
730 auto previous = value_from(-1);
731 auto current = value_from(this->to_string().size() / 2 - 1).pow10();
732
733 while (current.subtract(previous).abs().compare(1) == std::strong_ordering::greater) {
734 previous = current;
735 current = this->divide(previous).add(previous).divide(2);
736 }
737 return std::move(current);
738 }

745 [[nodiscard]]
746 big_integer gcd(const big_integer& rhs) const {
747 auto abs_lhs = this->abs();
748 auto abs_rhs = rhs.abs();
749
750 if (abs_rhs.equal(0) || abs_lhs.equal(0))
751 return 0;
752
753 auto remainder{abs_rhs};
754
755 while (!remainder.equal(0)) {
756 remainder = abs_lhs.mod(abs_rhs);
757 abs_lhs = abs_rhs; // previous remainder
758 abs_rhs = remainder; // current remainder
759 }
760 return std::move(abs_lhs);
761 }

931 std::stringstream ss;
