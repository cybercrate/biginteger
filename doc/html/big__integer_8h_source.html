359 [[nodiscard]]
360 std::strong_ordering compare(const big_integer& rhs) const
361 {
362 // -a, +b
363 if (this->is_negative() && rhs.is_positive())
364 return std::strong_ordering::less;
365
366 // +a, -b
367 if (this->is_positive() && rhs.is_negative())
368 return std::strong_ordering::greater;
369
370 // -a, -b
371 if (this->is_negative() && rhs.is_negative()) {
372 if (this->value_.length() == rhs.value_.length()) {
373 if (this->value_ > rhs.value_)
374 return std::strong_ordering::less;
375
376 return (this->value_ == rhs.value_)
377 ? std::strong_ordering::equivalent
378 : std::strong_ordering::greater;
379 }
380 return (this->value_.length() < rhs.value_.length())
381 ? std::strong_ordering::greater
382 : std::strong_ordering::less;
383 }
384
385 // +a, +b
386 if (this->value_.length() == rhs.value_.length()) {
387 if (this->value_ > rhs.value_)
388 return std::strong_ordering::greater;
389
390 return (this->value_ == rhs.value_)
391 ? std::strong_ordering::equivalent
392 : std::strong_ordering::less;
393 }
394 return (this->value_.length() < rhs.value_.length())
395 ? std::strong_ordering::less
396 : std::strong_ordering::greater;

439 [[nodiscard]]
440 big_integer add(const big_integer& rhs) const
441 {
442 // (-a)+(+b)
443 if (this->is_negative() && rhs.is_positive())
444 return rhs.subtract(this->negate());
445
446 // (+a)+(-b)
447 if (this->is_positive() && rhs.is_negative())
448 return this->subtract(rhs.negate());
449
450 // (+a)+(+b) or (-a)+(-b)
451 auto lhs_value = this->value_;
452 auto rhs_value = rhs.value_;
453
454 auto diff_length = std::abs(static_cast<int>(lhs_value.length() - rhs_value.length()));
455
456 if (lhs_value.length() > rhs_value.length())
457 rhs_value.insert(0, diff_length, '0');
458 else
459 lhs_value.insert(0, diff_length, '0');
460
461 std::ranges::reverse(lhs_value);
462 std::ranges::reverse(rhs_value);
463
464 std::size_t index{};
465 char carry{'0'};
466
467 for (char& lhs_char : lhs_value) {
468 lhs_char = static_cast<char>(
469 (carry - '0') + (lhs_char - '0') + (rhs_value.at(index) - '0') + '0');
470
471 if (lhs_char > '9') {
472 lhs_char -= 10;
473 carry = '1';
474 }
475 else {
476 carry = '0';
477 }
478 ++index;
479 }
480
481 if (carry > '0')
482 lhs_value.append(1, carry);
483
484 std::ranges::reverse(lhs_value);
485
486 return this->is_negative() ? value_from(lhs_value).negate() : std::move(lhs_value);
487 }

494 [[nodiscard]]
495 big_integer subtract(const big_integer& rhs) const
496 {
497 // (-a)-(+b) or (+a)-(-b)
498 if ((this->is_negative() && rhs.is_positive()) ||
499 (this->is_positive() && rhs.is_negative()))
500 {
501 return this->add(rhs.negate());
502 }
503
504 // (+a)-(+b) or (-a)-(-b)
505 if (this->is_negative())
506 return this->add(rhs.negate());
507
508 bool inverted_sign = (this->compare(rhs) == std::strong_ordering::less);
509
510 std::string subtracted = inverted_sign ? rhs.value_ : this->value_;
511 std::string removed = inverted_sign ? this->value_ : rhs.value_;
512
513 auto diff_length = std::abs(static_cast<int>(subtracted.length() - removed.length()));
514
515 if (subtracted.size() > removed.size())
516 removed.insert(0, diff_length, '0');
517 else
518 subtracted.insert(0, diff_length, '0');
519
520 std::ranges::reverse(subtracted);
521 std::ranges::reverse(removed);
522
523 std::size_t index{};
524
525 for (char& symbol : subtracted) {
526 if (symbol < removed.at(index)) {
527 symbol += 10;
528 subtracted[index + 1]--;
529 }
530 symbol = static_cast<char>((symbol - '0') - (removed.at(index) - '0') + '0');
531 ++index;
532 }
533 std::ranges::reverse(subtracted);
534
535 while (subtracted.front() == '0' && subtracted.length() != 1)
536 subtracted.erase(0, 1);
537
538 return inverted_sign ? value_from(subtracted).negate() : std::move(subtracted);
539 }

546 [[nodiscard]]
547 big_integer multiply(const big_integer& rhs) const
548 {
549 auto lhs_value = this->value_;
550 auto rhs_value = rhs.value_;
551
552 std::ranges::reverse(lhs_value);
553 std::ranges::reverse(rhs_value);
554
555 int index{};
556 char carry{'0'};
557
558 auto temp = value_from(0);
559
560 for (const auto& lhs_char : lhs_value) {
561 auto operation = std::string{}.insert(0, index, '0');
562
563 for (const auto& rhs_char : rhs_value) {
564 std::uint8_t result = ((lhs_char - '0') * (rhs_char - '0')) + (carry - '0') + '0';
565 carry = '0';
566
567 if (result > '9') {
568 while (result > '9') {
569 result -= 10;
570 ++carry;
571 }
572 }
573 operation.insert(0, 1, static_cast<char>(result));
574 }
575
576 if (carry > '0') {
577 operation.insert(0, 1, carry);
578 carry = '0';
579 }
580 temp += operation;
581 ++index;
582 }
583 auto positive = (this->is_negative() && rhs.is_negative()) ||
584 (this->is_positive() && rhs.is_positive());
585
586 return positive ? std::move(temp) : temp.negate();
587 }

594 [[nodiscard]]
595 big_integer divide(const big_integer& rhs) const
596 {
597 if (rhs.equal(0))
598 throw std::invalid_argument("divide by zero");
599
600 if (rhs.equal(1))
601 return *this;
602
603 if (this->equal(rhs))
604 return 1;
605
606 auto lhs_value = this->value_;
607 std::ranges::reverse(lhs_value);
608
609 auto rhs_abs = rhs.abs();
610
611 std::string lhs_quotient;
612 std::string rhs_quotient;
613
614 do {
615 lhs_quotient.push_back(lhs_value.back());
616 lhs_value.pop_back();
617
618 big_integer dividend_value{lhs_quotient};
619
620 if (dividend_value < rhs_abs) {
621 rhs_quotient.push_back('0');
622 continue;
623 }
624 auto number= value_from(2);
625
626 while (rhs_abs.multiply(number) <= dividend_value)
627 ++number;
628
629 --number;
630 rhs_quotient.append(number.to_string());
631 lhs_quotient = dividend_value.subtract(rhs_abs.multiply(number)).to_string();
632 } while (!lhs_value.empty());
633
634 bool positive = (this->is_negative() && rhs.is_negative()) ||
635 (this->is_positive() && rhs.is_positive());
636
637 return positive ? std::move(rhs_quotient) : value_from(rhs_quotient).negate();
638 }

740 [[nodiscard]]
741 big_integer pow(const big_integer& rhs) const
742 {
743 if (rhs.equal(0))
744 return 1;
745
746 if (this->equal(0))
747 return 0;
748
749 if (rhs.compare(0) == std::strong_ordering::less)
750 return this->abs().equal(1) ? *this : 0;
751
752 auto init_exp{rhs};
753 auto result{*this};
754 auto result_odd = value_from(1);
755
756 while (init_exp.compare(1) == std::strong_ordering::greater) {
757 if (init_exp.mod(2))
758 result_odd *= result;
759
760 result *= result;
761 init_exp /= 2;
762 }
763 return result.multiply(result_odd);
764 }

778 [[nodiscard]]
779 big_integer sqrt() const
780 {
781 if (this->compare(0) == std::strong_ordering::less)
782 throw std::invalid_argument("sqrt of a negative value");
783
784 // Small numbers optimisation.
785 if (this->equal(0))
786 return 0;
787
788 if (this->compare(4) == std::strong_ordering::less)
789 return 1;
790
791 if (this->compare(9) == std::strong_ordering::less)
792 return 2;
793
794 if (this->compare(16) == std::strong_ordering::less)
795 return 3;
796
797 auto previous = value_from(-1);
798 auto current = value_from(this->to_string().size() / 2 - 1).pow10();
799
800 while (current.subtract(previous).abs().compare(1) == std::strong_ordering::greater) {
801 previous = current;
802 current = this->divide(previous).add(previous).divide(2);
803 }
804 return std::move(current);
805 }

813 big_integer gcd(const big_integer& rhs) const
814 {

816 auto rhs_abs = rhs.abs();
817
818 if (rhs_abs.equal(0) || lhs_abs.equal(0))
819 return 0;
820
821 auto remainder{rhs_abs};
822
823 while (!remainder.equal(0)) {
824 remainder = lhs_abs.mod(rhs_abs);
825 lhs_abs = rhs_abs;
826 rhs_abs = remainder;
827 }
828 return std::move(lhs_abs);

1062 // Converts string value to base ten.
1063 static std::string convert_to_base_ten(const std::string& value, radix_type radix)
1064 {
1065 char last_char = value.back();
1066 int last = ('0' <= last_char) && (last_char <= '9')
1067 ? (last_char - '0')
1068 : (tolower(last_char) - 'a') + 10;
1069
1070 auto value_length = static_cast<std::size_t>(value.length());
1071
1072 auto power = value_from(value_length - 1);
1073 auto converted_value = value_from(last);
1074 auto radix_value = value_from(static_cast<int>(radix));
1075
1076 for (int i = 0; i < value_length - 1; ++i) {
1077 auto c = value.at(i);
1078 auto current_value = ('0' <= c) && (c <= '9')
1079 ? (c - '0')
1080 : (std::tolower(c) - 'a') + 10;
1081
1082 converted_value += value_from(current_value).multiply(radix_value.pow(power--));
1083 }
1084 return std::move(converted_value.value_);
1085 }

1087 // Converts not base ten value to string.
1088 static std::string convert_from_base_ten(const big_integer& value, const radix_type radix)
1089 {
1090 auto decimal_value{value};
1091 auto modulo = value_from(static_cast<std::uint8_t>(radix));
1092
1093 std::string result;
1094
1095 while (!decimal_value.equal(0)) {
1096 auto remainder = decimal_value.mod(modulo);
1097 decimal_value /= modulo;
1098
1099 auto symbol = base_chars().at(std::stoi(remainder.to_string()));
1100 result.push_back(symbol);
1101 }
1102 std::ranges::reverse(result);
1103
