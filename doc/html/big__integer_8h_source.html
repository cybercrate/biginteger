365 [[nodiscard]]
366 std::strong_ordering compare(const big_integer& rhs) const
367 {
368 // -a, +b
369 if (this->is_negative() && rhs.is_positive())
370 return std::strong_ordering::less;
371
372 // +a, -b
373 if (this->is_positive() && rhs.is_negative())
374 return std::strong_ordering::greater;
375
376 // -a, -b
377 if (this->is_negative() && rhs.is_negative()) {
378 if (this->value_.length() == rhs.value_.length()) {
379 if (this->value_ > rhs.value_)
380 return std::strong_ordering::less;
381
382 return (this->value_ == rhs.value_)
383 ? std::strong_ordering::equivalent
384 : std::strong_ordering::greater;
385 }
386 return (this->value_.length() < rhs.value_.length())
387 ? std::strong_ordering::greater
388 : std::strong_ordering::less;
389 }
390
391 // +a, +b
392 if (this->value_.length() == rhs.value_.length()) {
393 if (this->value_ > rhs.value_)
394 return std::strong_ordering::greater;
395
396 return (this->value_ == rhs.value_)
397 ? std::strong_ordering::equivalent
398 : std::strong_ordering::less;
399 }
400 return (this->value_.length() < rhs.value_.length())
401 ? std::strong_ordering::less
402 : std::strong_ordering::greater;
403 }

445 [[nodiscard]]
446 big_integer add(const big_integer& rhs) const
447 {
448 // (-a)+(+b)
449 if (this->is_negative() && rhs.is_positive())
450 return rhs.subtract(this->negate());
451
452 // (+a)+(-b)
453 if (this->is_positive() && rhs.is_negative())
454 return this->subtract(rhs.negate());
455
456 // (+a)+(+b) or (-a)+(-b)
457 auto lhs_value = this->value_;
458 auto rhs_value = rhs.value_;
459
460 auto diff_length = std::abs(static_cast<int>(lhs_value.length() - rhs_value.length()));
461
462 const auto min_value{'0'};
463 const auto max_value{'9'};
464
465 if (lhs_value.length() > rhs_value.length())
466 rhs_value.insert(0, diff_length, min_value);
467 else
468 lhs_value.insert(0, diff_length, min_value);
469
470 std::ranges::reverse(lhs_value);
471 std::ranges::reverse(rhs_value);
472
473 std::size_t index{};
474 char carry{min_value};
475
476 for (char& lhs_char : lhs_value) {
477 lhs_char = static_cast<char>(
478 (carry - min_value) + (lhs_char - min_value) + (rhs_value.at(index) - min_value) +
479 min_value);
480
481 if (lhs_char > max_value) {
482 lhs_char -= 10;
483 carry = (min_value + 1);
484 }
485 else {
486 carry = min_value;
487 }
488 ++index;
489 }
490
491 if (carry > min_value)
492 lhs_value.append(1, carry);
493
494 std::ranges::reverse(lhs_value);
495
496 return this->is_negative() ? value_from(lhs_value).negate() : std::move(lhs_value);
497 }

504 [[nodiscard]]
505 big_integer subtract(const big_integer& rhs) const
506 {
507 // (-a)-(+b) or (+a)-(-b)
508 if ((this->is_negative() && rhs.is_positive()) ||
509 (this->is_positive() && rhs.is_negative()))
510 {
511 return this->add(rhs.negate());
512 }
513
514 // (+a)-(+b) or (-a)-(-b)
515 if (this->is_negative())
516 return this->add(rhs.negate());
517
518 bool inverted_sign = (this->compare(rhs) == std::strong_ordering::less);
519
520 std::string subtracted = inverted_sign ? rhs.value_ : this->value_;
521 std::string removed = inverted_sign ? this->value_ : rhs.value_;
522
523 auto diff_length = std::abs(static_cast<int>(subtracted.length() - removed.length()));
524
525 const auto min_value{'0'};
526
527 if (subtracted.size() > removed.size())
528 removed.insert(0, diff_length, min_value);
529 else
530 subtracted.insert(0, diff_length, min_value);
531
532 std::ranges::reverse(subtracted);
533 std::ranges::reverse(removed);
534
535 std::size_t index{};
536
537 for (char& symbol : subtracted) {
538 if (symbol < removed.at(index)) {
539 symbol += 10;
540 --subtracted.at(index + 1);
541 }
542 symbol = static_cast<char>(
543 (symbol - min_value) - (removed.at(index) - min_value) + min_value);
544
545 ++index;
546 }
547 std::ranges::reverse(subtracted);
548
549 while (subtracted.front() == min_value && subtracted.length() != 1)
550 subtracted.erase(0, 1);
551
552 return inverted_sign ? value_from(subtracted).negate() : std::move(subtracted);
553 }

560 [[nodiscard]]
561 big_integer multiply(const big_integer& rhs) const
562 {
563 auto lhs_value = this->value_;
564 auto rhs_value = rhs.value_;
565
566 std::ranges::reverse(lhs_value);
567 std::ranges::reverse(rhs_value);
568
569 const auto min_value{'0'};
570 const auto max_value{'9'};
571
572 int index{};
573 char carry{min_value};
574
575 auto temp = value_from(0);
576
577 for (const auto& lhs_char : lhs_value) {
578 auto operation = std::string{}.insert(0, index, min_value);
579
580 for (const auto& rhs_char : rhs_value) {
581 std::uint8_t result = ((lhs_char - min_value) * (rhs_char - min_value)) +
582 (carry - min_value) + min_value;
583 carry = min_value;
584
585 if (result > max_value) {
586 while (result > max_value) {
587 result -= 10;
588 ++carry;
589 }
590 }
591 operation.insert(0, 1, static_cast<char>(result));
592 }
593
594 if (carry > min_value) {
595 operation.insert(0, 1, carry);
596 carry = min_value;
597 }
598 temp += operation;
599 ++index;
600 }
601 auto positive = (this->is_negative() && rhs.is_negative()) ||
602 (this->is_positive() && rhs.is_positive());
603
604 return positive ? std::move(temp) : temp.negate();
605 }

612 [[nodiscard]]
613 big_integer divide(const big_integer& rhs) const
614 {
615 if (rhs.equal(0))
616 throw std::invalid_argument("divide by zero");
617
618 if (rhs.equal(1))
619 return *this;
620
621 if (this->equal(rhs))
622 return 1;
623
624 auto lhs_value = this->value_;
625 std::ranges::reverse(lhs_value);
626
627 auto rhs_abs = rhs.abs();
628
629 std::string lhs_quotient;
630 std::string rhs_quotient;
631
632 do {
633 lhs_quotient.push_back(lhs_value.back());
634 lhs_value.pop_back();
635
636 big_integer dividend_value{lhs_quotient};
637
638 if (dividend_value < rhs_abs) {
639 rhs_quotient.push_back('0');
640 continue;
641 }
642 auto number= value_from(2);
643
644 while (rhs_abs.multiply(number) <= dividend_value)
645 ++number;
646
647 --number;
648 rhs_quotient.append(number.to_string());
649 lhs_quotient = dividend_value.subtract(rhs_abs.multiply(number)).to_string();
650 } while (!lhs_value.empty());
651
652 bool positive = (this->is_negative() && rhs.is_negative()) ||
653 (this->is_positive() && rhs.is_positive());
654
655 return positive ? std::move(rhs_quotient) : value_from(rhs_quotient).negate();
656 }

758 [[nodiscard]]
759 big_integer pow(const big_integer& rhs) const
760 {
761 if (rhs.equal(0))
762 return 1;
763
764 if (this->equal(0))
765 return 0;
766
767 if (rhs.compare(0) == std::strong_ordering::less)
768 return this->abs().equal(1) ? *this : 0;
769
770 auto init_exp{rhs};
771 auto result{*this};
772 auto result_odd = value_from(1);
773
774 while (init_exp.compare(1) == std::strong_ordering::greater) {
775 if (init_exp.mod(2))
776 result_odd *= result;
777
778 result *= result;
779 init_exp /= 2;
780 }
781 return result.multiply(result_odd);
782 }

796 [[nodiscard]]
797 big_integer sqrt() const
798 {
799 if (this->compare(0) == std::strong_ordering::less)
800 throw std::invalid_argument("sqrt of a negative value");
801
802 // Small numbers optimisation.
803 if (this->equal(0))
804 return 0;
805
806 if (this->compare(4) == std::strong_ordering::less)
807 return 1;
808
809 if (this->compare(9) == std::strong_ordering::less)
810 return 2;
811
812 if (this->compare(16) == std::strong_ordering::less)
813 return 3;
814
815 auto previous = value_from(-1);
816 auto current = value_from(this->to_string().size() / 2 - 1).pow10();
817
818 while (current.subtract(previous).abs().compare(1) == std::strong_ordering::greater) {
819 previous = current;
820 current = this->divide(previous).add(previous).divide(2);
821 }
822 return current;
823 }

831 big_integer gcd(const big_integer& rhs) const
832 {
833 auto lhs_abs = this->abs();
834 auto rhs_abs = rhs.abs();
835
836 if (rhs_abs.equal(0) || lhs_abs.equal(0))
837 return 0;
838

840
841 while (!remainder.equal(0)) {

843 lhs_abs = rhs_abs;
