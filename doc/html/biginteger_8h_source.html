254 [[nodiscard]]
255 std::strong_ordering compare(const biginteger& rhs) const {
256 // -a, +b
257 if (this->is_negative() && rhs.is_positive()) {
258 return std::strong_ordering::less;
259 }
260
261 // +a, -b
262 if (this->is_positive() && rhs.is_negative()) {
263 return std::strong_ordering::greater;
264 }
265
266 // -a, -b
267 if (this->is_negative() && rhs.is_negative()) {
268 if (this->value_.length() == rhs.value_.length()) {
269 if (this->value_ > rhs.value_) {
270 return std::strong_ordering::less;
271 }
272 return (this->value_ == rhs.value_)
273 ? std::strong_ordering::equivalent
274 : std::strong_ordering::greater;
275 }
276 return (this->value_.length() < rhs.value_.length())
277 ? std::strong_ordering::greater
278 : std::strong_ordering::less;
279 }
280
281 // +a, +b
282 if (this->value_.length() == rhs.value_.length()) {
283 if (this->value_ > rhs.value_) {
284 return std::strong_ordering::greater;
285 }
286 return (this->value_ == rhs.value_)
287 ? std::strong_ordering::equivalent
288 : std::strong_ordering::less;
289 }
290 return (this->value_.length() < rhs.value_.length())
291 ? std::strong_ordering::less
292 : std::strong_ordering::greater;

312 [[nodiscard]]
313 biginteger add(const biginteger& rhs) const {
314 // (-a)+(+b)
315 if (this->is_negative() && rhs.is_positive()) {
316 return rhs.subtract(this->negate());
317 }
318
319 // (+a)+(-b)
320 if (this->is_positive() && rhs.is_negative()) {
321 return this->subtract(rhs.negate());
322 }
323
324 // (+a)+(+b) or (-a)+(-b)
325 auto lhs_value = this->value_;
326 auto rhs_value = rhs.value_;
327
328 auto diff_length = std::abs(
329 static_cast<int>(lhs_value.length() - rhs_value.length()));
330
331 const auto min_value{'0'};
332 const auto max_value{'9'};
333
334 if (lhs_value.length() > rhs_value.length()) {
335 rhs_value.insert(0, diff_length, min_value);
336 } else {
337 lhs_value.insert(0, diff_length, min_value);
338 }
339 std::ranges::reverse(lhs_value);
340 std::ranges::reverse(rhs_value);
341
342 usize index{};
343 char carry{min_value};
344
345 for (char& lhs_char : lhs_value) {
346 lhs_char = static_cast<char>(
347 (carry - min_value) +
348 (lhs_char - min_value) +
349 (rhs_value.at(index) - min_value) +
350 min_value);
351
352 if (lhs_char > max_value) {
353 lhs_char -= 10;
354 carry = (min_value + 1);
355 } else {
356 carry = min_value;
357 }
358 ++index;
359 }
360
361 if (carry > min_value) {
362 lhs_value.append(1, carry);
363 }
364 std::ranges::reverse(lhs_value);
365
366 return this->is_negative() ? value_from(lhs_value).negate() : std::move(lhs_value);
367 }

374 [[nodiscard]]
375 biginteger subtract(const biginteger& rhs) const {
376 // (-a)-(+b) or (+a)-(-b)
377 if ((this->is_negative() && rhs.is_positive()) ||
378 (this->is_positive() && rhs.is_negative()))
379 {
380 return this->add(rhs.negate());
381 }
382
383 // (+a)-(+b) or (-a)-(-b)
384 if (this->is_negative()) {
385 return this->add(rhs.negate());
386 }
387 bool inverted_sign = (this->compare(rhs) == std::strong_ordering::less);
388
389 std::string subtracted = inverted_sign ? rhs.value_ : this->value_;
390 std::string removed = inverted_sign ? this->value_ : rhs.value_;
391
392 auto diff_length = std::abs(
393 static_cast<int>(subtracted.length() - removed.length()));
394
395 const auto min_value{'0'};
396
397 if (subtracted.size() > removed.size()) {
398 removed.insert(0, diff_length, min_value);
399 } else {
400 subtracted.insert(0, diff_length, min_value);
401 }
402 std::ranges::reverse(subtracted);
403 std::ranges::reverse(removed);
404
405 usize index{};
406
407 for (char& symbol : subtracted) {
408 if (symbol < removed.at(index)) {
409 symbol += 10;
410 --subtracted.at(index + 1);
411 }
412 symbol = static_cast<char>(
413 (symbol - min_value) - (removed.at(index) - min_value) + min_value);
414
415 ++index;
416 }
417 std::ranges::reverse(subtracted);
418
419 while (subtracted.front() == min_value && subtracted.length() != 1) {
420 subtracted.erase(0, 1);
421 }
422 return inverted_sign ? value_from(subtracted).negate() : std::move(subtracted);
423 }

430 [[nodiscard]]
431 biginteger multiply(const biginteger& rhs) const {
432 auto lhs_value = this->value_;
433 auto rhs_value = rhs.value_;
434
435 std::ranges::reverse(lhs_value);
436 std::ranges::reverse(rhs_value);
437
438 const auto min_value{'0'};
439 const auto max_value{'9'};
440
441 int index{};
442 char carry{min_value};
443
444 auto temp = value_from(0);
445 uint8 result;
446
447 for (const auto& lhs_char : lhs_value) {
448 auto operation = std::string{}.insert(0, index, min_value);
449
450 for (const auto& rhs_char : rhs_value) {
451 result =
452 ((lhs_char - min_value) * (rhs_char - min_value)) +
453 (carry - min_value) +
454 min_value;
455
456 carry = min_value;
457
458 if (result > max_value) {
459 while (result > max_value) {
460 result -= 10;
461 ++carry;
462 }
463 }
464 operation.insert(0, 1, static_cast<char>(result));
465 }
466 if (carry > min_value) {
467 operation.insert(0, 1, carry);
468 carry = min_value;
469 }
470 temp = temp.add(operation);
471 ++index;
472 }
473 auto positive =
474 (this->is_negative() && rhs.is_negative()) ||
475 (this->is_positive() && rhs.is_positive());
476
477 return positive ? std::move(temp) : temp.negate();
478 }

485 [[nodiscard]]
486 biginteger divide(const biginteger& rhs) const {
487 if (rhs.equal(0)) {
488 throw std::invalid_argument("divide by zero");
489 }
490 if (rhs.equal(1)) {
491 return *this;
492 }
493 if (this->equal(rhs)) {
494 return 1;
495 }
496 auto lhs_value = this->value_;
497 std::ranges::reverse(lhs_value);
498
499 auto rhs_abs = rhs.abs();
500
501 std::string lhs_quotient;
502 std::string rhs_quotient;
503
504 do {
505 lhs_quotient.push_back(lhs_value.back());
506 lhs_value.pop_back();
507
508 biginteger dividend_value{lhs_quotient};
509
510 if (dividend_value.compare(rhs_abs) == std::strong_ordering::less) {
511 rhs_quotient.push_back('0');
512 continue;
513 }
514 auto number = value_from(2);
515 auto mul = rhs_abs.multiply(number).compare(dividend_value);
516
517 while (mul == std::strong_ordering::less || mul == std::strong_ordering::equal) {
518 ++number;
519 mul = rhs_abs.multiply(number).compare(dividend_value);
520 }
521 --number;
522 rhs_quotient.append(number.to_string());
523 lhs_quotient = dividend_value.subtract(rhs_abs.multiply(number)).to_string();
524 } while (!lhs_value.empty());
525
526 auto positive =
527 (this->is_negative() && rhs.is_negative()) ||
528 (this->is_positive() && rhs.is_positive());
529
530 return positive ? std::move(rhs_quotient) : value_from(rhs_quotient).negate();
531 }

578 [[nodiscard]]
579 biginteger pow(const biginteger& rhs) const {
580 if (rhs.equal(0)) {
581 return 1;
582 }
583 if (this->equal(0)) {
584 return 0;
585 }
586 if (rhs.compare(0) == std::strong_ordering::less) {
587 return this->abs().equal(1) ? *this : 0;
588 }
589 auto init_exp{rhs};
590 auto result{*this};
591 auto result_odd = value_from(1);
592
593 while (init_exp.compare(1) == std::strong_ordering::greater) {
594 if (init_exp.mod(2)) {
595 result_odd = result_odd.multiply(result);
596 }
597 result = result.multiply(result);
598 init_exp = init_exp.divide(2);
599 }
600 return result.multiply(result_odd);
601 }

614 [[nodiscard]]
615 biginteger sqrt() const {
616 if (this->compare(0) == std::strong_ordering::less) {
617 throw std::invalid_argument("sqrt of a negative value");
618 }
619
620 // Small numbers optimisation.
621 if (this->equal(0)) {
622 return 0;
623 }
624 if (this->compare(4) == std::strong_ordering::less) {
625 return 1;
626 }
627 if (this->compare(9) == std::strong_ordering::less) {
628 return 2;
629 }
630 if (this->compare(16) == std::strong_ordering::less) {
631 return 3;
632 }
633 auto previous = value_from(-1);
634 auto current = value_from(this->to_string().size() / 2 - 1).pow10();
635
636 while (current.subtract(previous).abs().compare(1) == std::strong_ordering::greater) {
637 previous = current;
638 current = this->divide(previous).add(previous).divide(2);
639 }
640 return current;
641 }

648 [[nodiscard]]
649 biginteger gcd(const biginteger& rhs) const {
650 auto lhs_abs = this->abs();
651 auto rhs_abs = rhs.abs();
652
653 if (rhs_abs.equal(0) || lhs_abs.equal(0)) {
654 return 0;
655 }
656 auto remainder{rhs_abs};
657
658 while (!remainder.equal(0)) {
659 remainder = lhs_abs.mod(rhs_abs);
660 lhs_abs = rhs_abs;
661 rhs_abs = remainder;
662 }
663 return lhs_abs;
664 }

803 static std::string convert_from_base_ten(const biginteger& value, const radix_flag radix) {
804 auto decimal_value{value};
805 auto modulo = value_from(static_cast<uint8>(radix));
806
807 std::string result;
808

810 auto remainder = decimal_value.mod(modulo);
811 decimal_value = decimal_value.divide(modulo);
812
813 auto c = base_chars(radix_flag::hexadecimal).at(std::stoi(remainder.to_string()));

821 // Converts string value to base ten.
822 static std::string convert_to_base_ten(const std::string& value, radix_flag radix) {
823 auto last = char_to_int32(value.back());
824 auto length = value.length();
825
826 auto power = value_from(length - 1);
827 auto converted_value = value_from(last);
828 auto radix_value = value_from(static_cast<uint8>(radix));
829
830 int current;
831
832 for (usize i = 0; i < length - 1; ++i) {
833 current = char_to_int32(value.at(i));
834 converted_value = converted_value
835 .add(value_from(current).multiply(radix_value.pow(power--)));
