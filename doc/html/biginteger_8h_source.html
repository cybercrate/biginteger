212 if (rhs.is_negative()) {

255 [[nodiscard]]
256 std::strong_ordering compare(const biginteger& rhs) const {
257 // -a, +b
258 if (this->is_negative() && rhs.is_positive()) {
259 return std::strong_ordering::less;
260 }
261
262 // +a, -b
263 if (this->is_positive() && rhs.is_negative()) {
264 return std::strong_ordering::greater;
265 }
266
267 // -a, -b
268 if (this->is_negative() && rhs.is_negative()) {
269 if (this->value_.length() == rhs.value_.length()) {
270 if (this->value_ > rhs.value_) {
271 return std::strong_ordering::less;
272 }
273 return (this->value_ == rhs.value_)
274 ? std::strong_ordering::equivalent
275 : std::strong_ordering::greater;
276 }
277 return (this->value_.length() < rhs.value_.length())
278 ? std::strong_ordering::greater
279 : std::strong_ordering::less;
280 }
281
282 // +a, +b
283 if (this->value_.length() == rhs.value_.length()) {
284 if (this->value_ > rhs.value_) {
285 return std::strong_ordering::greater;
286 }
287 return (this->value_ == rhs.value_)
288 ? std::strong_ordering::equivalent
289 : std::strong_ordering::less;
290 }
291 return (this->value_.length() < rhs.value_.length())
292 ? std::strong_ordering::less
293 : std::strong_ordering::greater;

313 [[nodiscard]]
314 biginteger add(const biginteger& rhs) const {
315 // (-a)+(+b)
316 if (this->is_negative() && rhs.is_positive()) {
317 return rhs.subtract(this->negate());
318 }
319
320 // (+a)+(-b)
321 if (this->is_positive() && rhs.is_negative()) {
322 return this->subtract(rhs.negate());
323 }
324
325 // (+a)+(+b) or (-a)+(-b)
326 auto lhs_value = this->value_;
327 auto rhs_value = rhs.value_;
328
329 auto diff_length = std::abs(
330 static_cast<int>(lhs_value.length() - rhs_value.length()));
331
332 const auto min_value{'0'};
333 const auto max_value{'9'};
334
335 if (lhs_value.length() > rhs_value.length()) {
336 rhs_value.insert(0, diff_length, min_value);
337 } else {
338 lhs_value.insert(0, diff_length, min_value);
339 }
340 std::ranges::reverse(lhs_value);
341 std::ranges::reverse(rhs_value);
342
343 std::size_t index{};
344 char carry{min_value};
345
346 for (char& lhs_char : lhs_value) {
347 lhs_char = static_cast<char>(
348 (carry - min_value) +
349 (lhs_char - min_value) +
350 (rhs_value.at(index) - min_value) +
351 min_value);
352
353 if (lhs_char > max_value) {
354 lhs_char -= 10;
355 carry = (min_value + 1);
356 } else {
357 carry = min_value;
358 }
359 ++index;
360 }
361
362 if (carry > min_value) {
363 lhs_value.append(1, carry);
364 }
365 std::ranges::reverse(lhs_value);
366
367 return this->is_negative() ? value_from(lhs_value).negate() : std::move(lhs_value);
368 }

375 [[nodiscard]]
376 biginteger subtract(const biginteger& rhs) const {
377 // (-a)-(+b) or (+a)-(-b)
378 if ((this->is_negative() && rhs.is_positive()) ||
379 (this->is_positive() && rhs.is_negative()))
380 {
381 return this->add(rhs.negate());
382 }
383
384 // (+a)-(+b) or (-a)-(-b)
385 if (this->is_negative()) {
386 return this->add(rhs.negate());
387 }
388 bool inverted_sign = (this->compare(rhs) == std::strong_ordering::less);
389
390 std::string subtracted = inverted_sign ? rhs.value_ : this->value_;
391 std::string removed = inverted_sign ? this->value_ : rhs.value_;
392
393 auto diff_length = std::abs(
394 static_cast<int>(subtracted.length() - removed.length()));
395
396 const auto min_value{'0'};
397
398 if (subtracted.size() > removed.size()) {
399 removed.insert(0, diff_length, min_value);
400 } else {
401 subtracted.insert(0, diff_length, min_value);
402 }
403 std::ranges::reverse(subtracted);
404 std::ranges::reverse(removed);
405
406 std::size_t index{};
407
408 for (char& symbol : subtracted) {
409 if (symbol < removed.at(index)) {
410 symbol += 10;
411 --subtracted.at(index + 1);
412 }
413 symbol = static_cast<char>(
414 (symbol - min_value) - (removed.at(index) - min_value) + min_value);
415
416 ++index;
417 }
418 std::ranges::reverse(subtracted);
419
420 while (subtracted.front() == min_value && subtracted.length() != 1) {
421 subtracted.erase(0, 1);
422 }
423 return inverted_sign ? value_from(subtracted).negate() : std::move(subtracted);
424 }

431 [[nodiscard]]
432 biginteger multiply(const biginteger& rhs) const {
433 auto lhs_value = this->value_;
434 auto rhs_value = rhs.value_;
435
436 std::ranges::reverse(lhs_value);
437 std::ranges::reverse(rhs_value);
438
439 const auto min_value{'0'};
440 const auto max_value{'9'};
441
442 int index{};
443 char carry{min_value};
444
445 auto temp = value_from(0);
446 std::uint8_t result;
447
448 for (const auto& lhs_char : lhs_value) {
449 auto operation = std::string{}.insert(0, index, min_value);
450
451 for (const auto& rhs_char : rhs_value) {
452 result =
453 ((lhs_char - min_value) * (rhs_char - min_value)) +
454 (carry - min_value) +
455 min_value;
456
457 carry = min_value;
458
459 if (result > max_value) {
460 while (result > max_value) {
461 result -= 10;
462 ++carry;
463 }
464 }
465 operation.insert(0, 1, static_cast<char>(result));
466 }
467 if (carry > min_value) {
468 operation.insert(0, 1, carry);
469 carry = min_value;
470 }
471 temp = temp.add(operation);
472 ++index;
473 }
474 auto positive =
475 (this->is_negative() && rhs.is_negative()) ||
476 (this->is_positive() && rhs.is_positive());
477
478 return positive ? std::move(temp) : temp.negate();
479 }

486 [[nodiscard]]
487 biginteger divide(const biginteger& rhs) const {
488 if (rhs.equal(0)) {
489 throw std::invalid_argument("divide by zero");
490 }
491 if (rhs.equal(1)) {
492 return *this;
493 }
494 if (this->equal(rhs)) {
495 return 1;
496 }
497 auto lhs_value = this->value_;
498 std::ranges::reverse(lhs_value);
499
500 auto rhs_abs = rhs.abs();
501
502 std::string lhs_quotient;
503 std::string rhs_quotient;
504
505 do {
506 lhs_quotient.push_back(lhs_value.back());
507 lhs_value.pop_back();
508
509 biginteger dividend_value{lhs_quotient};
510
511 if (dividend_value.compare(rhs_abs) == std::strong_ordering::less) {
512 rhs_quotient.push_back('0');
513 continue;
514 }
515 auto number = value_from(2);
516 auto mul = rhs_abs.multiply(number).compare(dividend_value);
517
518 while (mul == std::strong_ordering::less || mul == std::strong_ordering::equal) {
519 ++number;
520 mul = rhs_abs.multiply(number).compare(dividend_value);
521 }
522 --number;
523 rhs_quotient.append(number.to_string());
524 lhs_quotient = dividend_value.subtract(rhs_abs.multiply(number)).to_string();
525 } while (!lhs_value.empty());
526
527 auto positive =
528 (this->is_negative() && rhs.is_negative()) ||
529 (this->is_positive() && rhs.is_positive());
530
531 return positive ? std::move(rhs_quotient) : value_from(rhs_quotient).negate();
532 }

579 [[nodiscard]]
580 biginteger pow(const biginteger& rhs) const {
581 if (rhs.equal(0)) {
582 return 1;
583 }
584 if (this->equal(0)) {
585 return 0;
586 }
587 if (rhs.compare(0) == std::strong_ordering::less) {
588 return this->abs().equal(1) ? *this : 0;
589 }
590 auto init_exp{rhs};
591 auto result{*this};
592 auto result_odd = value_from(1);
593
594 while (init_exp.compare(1) == std::strong_ordering::greater) {
595 if (init_exp.mod(2)) {
596 result_odd = result_odd.multiply(result);
597 }
598 result = result.multiply(result);
599 init_exp = init_exp.divide(2);
600 }
601 return result.multiply(result_odd);
602 }

615 [[nodiscard]]
616 biginteger sqrt() const {
617 if (this->compare(0) == std::strong_ordering::less) {
618 throw std::invalid_argument("sqrt of a negative value");
619 }
620
621 // Small numbers optimisation.
622 if (this->equal(0)) {
623 return 0;
624 }
625 if (this->compare(4) == std::strong_ordering::less) {
626 return 1;
627 }
628 if (this->compare(9) == std::strong_ordering::less) {
629 return 2;
630 }
631 if (this->compare(16) == std::strong_ordering::less) {
632 return 3;
633 }
634 auto previous = value_from(-1);
635 auto current = value_from(this->to_string().size() / 2 - 1).pow10();
636
637 while (current.subtract(previous).abs().compare(1) == std::strong_ordering::greater) {
638 previous = current;
639 current = this->divide(previous).add(previous).divide(2);
640 }
641 return current;
642 }

649 [[nodiscard]]
650 biginteger gcd(const biginteger& rhs) const {
651 auto lhs_abs = this->abs();
652 auto rhs_abs = rhs.abs();
653
654 if (rhs_abs.equal(0) || lhs_abs.equal(0)) {
655 return 0;
656 }
657 auto remainder{rhs_abs};
658
659 while (!remainder.equal(0)) {
660 remainder = lhs_abs.mod(rhs_abs);
661 lhs_abs = rhs_abs;
662 rhs_abs = remainder;
663 }
664 return lhs_abs;
665 }

809 auto decimal_value{value};

826 // Converts string value to base ten.
827 static std::string convert_to_base_ten(const std::string& value, radix_flag radix) {
828 auto last = __detail::char_to_int32(value.back());
829 auto length = value.length();
830
831 auto power = value_from(length - 1);
832 auto converted_value = value_from(last);
833 auto radix_value = value_from(static_cast<std::uint8_t>(radix));
834
835 int current;
836
837 for (std::size_t i = 0; i < length - 1; ++i) {
838 current = __detail::char_to_int32(value.at(i));
839 converted_value = converted_value
840 .add(value_from(current).multiply(radix_value.pow(power--)));
841 }
842 return std::move(converted_value.value_);
843 }
